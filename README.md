#### merge 函数，合并任意两个对象为一个新对象，若两个对象有相同键名，则将其键值合并，放在一个数组中作为新对象相应键名下的键值。若两个对象中有其中一个对象没有的属性，该属性也应该出现在合并后的新对象中。
  function merge(obj1, obj2) {
    // ...
}

var obj1 = {
  a: [{ x: 2 }, { y: 4 }],
  b: 1,
  c: 'bar'
}

var obj2 = {
  a: { z: 3 },
  b: [2, 3],
  c: 'foo',
  d: 6
}

merge(obj1, obj2)

// { a: [ { x: 2 }, { y: 4 }, { z: 3 } ], b: [ 1, 2, 3 ], c: ['bar', 'foo'], d: 6 }
